        for(int j = i + 1; j <= n; j++){
            if(dis(a[i], a[j]) < 2. * x) d.join(i, j);
        }
    }
    int root = d.find(1);
    for(int i = 2; i <= n; i++) if(d.find(i) != root) return false;
    return true;
}
